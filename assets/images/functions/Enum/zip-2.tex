\newcommand{\zipelem}[1]{
    \elemx{#1}{$\{a_#1,b_#1\}$}
}

\matrix [fptk, list=a] {
    \elem{1} &
    \elem{2} &
    \elems &
    \elem{n} \\
};

\matrix [fptk, list=b, below=.5 of a] {
    \elem[b]{1} &
    \elem[b]{2} &
    \elems &  
    \elem[b]{n} \\
};


\matrix [fptk, list=c, below=1 of b, nodes={font=\small}] {
    \zipelem{1} &
    \zipelem{2} &
    \elems &
    \zipelem{n} \\
};


\foreach \i in {1, 2, n}{
    \draw [fptk, group] (a\i.center) -- (b\i.center);
    \draw [fptk, flow ->] (last group.east) -- (c\i);
}
