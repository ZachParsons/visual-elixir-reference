\makeatletter

\pgfkeys{
    fptk/.cd,
    exception line/.code={
        \pgfkeys{
            tikz/.cd,
            decorate,
            decoration={
                amplitude=.25mm,
                segment length=2mm,
                zigzag,
            },
        },
    },
    exception/.code={
        \pgfkeys{
            /tikz/.cd,
            draw,
            /fptk/exception line,
            line width=3\defaultpenwidth,
            node contents=\texttt{#1},
            tips=false,
        }
    },
    throw ->/.style={
        /fptk/flow ->=exception,
    },
    flow shape/exception/.style={
        /fptk/exception line,
    },
}

\makeatother
